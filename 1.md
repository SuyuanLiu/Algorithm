class Parenthesis:
    def chkParenthesis(self, A, n):
        # write code here
        if not A:
            return True
        if n % 2 == 1:
            return False
        
        s, i = [], 0
        while s or i < n:
            if A[i] not in ['(', ')']:
                return False
            if not s:
                s.append(A[i])
            else:
                if A[i] == ')':
                    tmp = s.pop()
                    if tmp == ')':
                        return False
                else:
                    s.append(A[i])
            i += 1
        return s == []

class LCS:
    def findLCS(self, A, n, B, m):
        # write code here
        if not A or not B:
            return 0
        dp = [[0 for j in range(n+1)] for i in range(m+1)]
        for i in range(1, m+1):
            for j in range(1, n+1):
                if A[j-1] == B[i-1]:
                    dp[i][j] = dp[i-1][j-1] + 1
                else:
                    dp[i][j] = max(dp[i-1][j], dp[i][j-1])
        return dp[-1][-1]

class GoUpstairs:
    def countWays(self, n):
        # write code here
        res = [1, 2, 4]
        if n < 4:
            return res[n-1]
        
        f1, f2, f3 = 1, 2, 4
        res = 0
        for i in range(3, n):
            res = f1 + f2 + f3 
            f1 = f2 
            f2 = f3
            f3 = res
        return res%1000000007

class NextElement:
    def helper(self, A, idx, n):
        res = float('inf')
        for i in range(idx+1, n):
            if A[i] > A[idx] and A[i] < res:
                res = A[i]
        return res if res != float('inf') else -1
    
    def findNext(self, A, n):
        # write code here
        if not A:
            return []
        
        res = []
        for i in range(n-1):
            res.append(self.helper(A, i, n))
        res.append(-1)
        return res
